-- VERİTABANI TABLO SÜTUN LİSTELEME

SELECT 
    TABLE_NAME,
    GROUP_CONCAT(CONCAT(COLUMN_NAME, ' (', DATA_TYPE, ')') ORDER BY ORDINAL_POSITION SEPARATOR ', ') AS COLUMNS
FROM 
    INFORMATION_SCHEMA.COLUMNS
WHERE 
    TABLE_SCHEMA = 'personel_takip'
GROUP BY 
    TABLE_NAME
ORDER BY 
    TABLE_NAME ASC;



SELECT TABLE_NAME, COLUMN_NAME, DATA_TYPE
FROM INFORMATION_SCHEMA.COLUMNS
WHERE TABLE_SCHEMA = 'personel_takip'
ORDER BY TABLE_NAME, ORDINAL_POSITION;
